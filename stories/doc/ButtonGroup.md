# ButtonGroup

## Properties

Name      | Type     | Description
--------- | -------- | -----------
className | string   | css class name of the element

## Usage

```js
import {Button, ButtonGroup} from 'react-pure-css'

render () {
  return (
    <ButtonGroup>
      <Button>Pure</Button>
      <Button primary>Primary</Button>
      <Button secondary>Secondary</Button>
    </ButtonGroup>
  )
}
```
